#  Практические примеры применения `enum`

##  Сравнение порядковых номеров

Можно сравнивать**` enum`** через **`ordinal`**:

enum Size:
case Small, Medium, Large, XLarge

def compare(s1: Size, s2: Size): Int =
s1.ordinal - s2.ordinal

println(compare(Size.Small, Size.Large)) // отрицательное
println(compare(Size.Large, Size.Small)) // положительное
println(compare(Size.Medium, Size.Medium)) // 0

## Перебор значений

### Итерация по всем значениям:

enum Day:
case Mon, Tue, Wed, Thu, Fri, Sat, Sun

Day.values.foreach { day =>
println(s"$day: ordinal ${day.ordinal}")
}

### Фильтрация:

enum Grade:
case A, B, C, D, F

val passingGrades = Grade.values.filter(_.ordinal < 4)
// Array(A, B, C, D)

### Поиск:

enum State:
case Init, Running, Paused, Finished

val running = State.values.find(_.ordinal == 1)
// Some(Running)

### Получение имен всех значений:

enum Command:
case Start, Stop, Restart

val names = Command.values.map(_.toString)
// Array("Start", "Stop", "Restart")

## Моделирование направления

enum Direction:
case North
case South
case East
case West

// |Direction| = 4

def opposite(d: Direction): Direction = d match
case Direction.North => Direction.South
case Direction.South => Direction.North
case Direction.East => Direction.West
case Direction.West => Direction.East

def turnRight(d: Direction): Direction = d match
case Direction.North => Direction.East
case Direction.East => Direction.South
case Direction.South => Direction.West
case Direction.West => Direction.North

// Проверка
println(opposite(Direction.North)) // South
println(turnRight(Direction.North)) // East

## HTTP методы

enum HttpMethod:
case GET
case POST
case PUT
case DELETE
case PATCH

// |HttpMethod| = 5

def isIdempotent(method: HttpMethod): Boolean = method match
case HttpMethod.GET | HttpMethod.PUT | HttpMethod.DELETE => true
case HttpMethod.POST | HttpMethod.PATCH => false

println(isIdempotent(HttpMethod.GET)) // true
println(isIdempotent(HttpMethod.POST)) // false

## Моделирование состояния процесса

enum GameState:
case MainMenu
case Playing
case Paused
case GameOver

// |GameState| = 4

def canSaveGame(state: GameState): Boolean = state match
case GameState.Playing => true
case GameState.Paused => true
case GameState.MainMenu | GameState.GameOver => false

## Команда с параметром

enum Command:
case Start
case Stop
case SetSpeed(value: Int)
case SetMode(mode: String)

def executeCommand(cmd: Command): Unit = cmd match
case Command.Start => println("Starting...")
case Command.Stop => println("Stopping...")
case Command.SetSpeed(v) => println(s"Speed set to $v")
case Command.SetMode(m) => println(s"Mode changed to $m")

// Использование
executeCommand(Command.Start)
executeCommand(Command.SetSpeed(100))
executeCommand(Command.SetMode("turbo"))

## Токен авторизации

enum AuthToken:
case Anonymous
case Bearer(token: String)
case ApiKey(key: String)

def getAuthHeader(token: AuthToken): Option[String] = token match
case AuthToken.Anonymous => None
case AuthToken.Bearer(t) => Some(s"Bearer $t")
case AuthToken.ApiKey(k) => Some(s"X-API-Key: $k")

// Примеры
println(getAuthHeader(AuthToken.Anonymous)) // None
println(getAuthHeader(AuthToken.Bearer("abc123"))) // Some(Bearer abc123)
println(getAuthHeader(AuthToken.ApiKey("secret"))) // Some(X-API-Key: secret)
