## Вывод диапазонов целых чисел

val value = 100.toByte
val result: String = value match
case b: Byte => s"Byte: $b (диапазон: ${Byte.MinValue} до ${Byte.MaxValue})"
case s: Short => s"Short: $s (диапазон: ${Short.MinValue} до ${Short.MaxValue})"
case i: Int => s"Int: $i (диапазон: ${Int.MinValue} до ${Int.MaxValue})"
case l: Long => s"Long: $l (диапазон: ${Long.MinValue} до ${Long.MaxValue})"
case _ => "Не числовой тип"

result // = "Byte: 100 (диапазон: -128 до 127)"

## Расшифровка типов числовых литералов

val value = 42.0f
val result: String = value match
case 42 => "Десятичное 42"
case 42L => "Long 42"
case 0x2A => "Шестнадцатеричное 42 (0x2A)"
case 0b101010 => "Двоичное 42 (0b101010)"
case 42.0 => "Double 42.0"
case 42.0f => "Float 42.0f"
case _ => s"Другое значение: $value"

result // = "Float 42.0f"

case _ => "Не число с плавающей точкой"

result // = "NaN (Double)"

## Классификация символов

val ch: Char = '!'
val result: String = ch match
case '\n' => "Перевод строки"
case '\t' => "Табуляция"
case '\r' => "Возврат каретки"
case '\\' => "Обратный слеш"
case '\'' => "Одинарная кавычка"
case c if c.isDigit => s"Цифра: $c (код: ${c.toInt})"
case c if c.isLetter => s"Буква: $c (код: ${c.toInt})"
case c if c.isWhitespace => s"Пробельный символ (код: ${c.toInt})"
case c => s"Знак пунктуации или специальный символ: $c (код: ${c.toInt})"

result // = "Знак пунктуации: !"

## Строчные или заглавные буквы

val result: String = ch match

case c if c >= 'a' && c <= 'z' => "Строчная латинская буква"
case c if c >= 'A' && c <= 'Z' => "Заглавная латинская буква"
case c if c >= 'а' && c <= 'я' => "Строчная кириллическая буква"
case c if c >= 'А' && c <= 'Я' => "Заглавная кириллическая буква"
case c => s"Другой символ: $c"

result // = "Заглавная кириллическая буква"

val value: Int = 42

value match

## Конвертер значений различных типов в строку

val value = true
val result: String = value match
case b: Byte => s"$b (byte)"
case s: Short => s"$s (short)"
case i: Int => s"$i (int)"
case l: Long => s"$l (long)"
case f: Float => s"$f (float)"
case d: Double => s"$d (double)"
case c: Char => s"'$c' (char, код: ${c.toInt})"
case b: Boolean => if (b) "истина" else "ложь"
case _ => "неизвестный тип"

result // = "истина"

## Арифметический калькулятор

val operation: Char = '+'
val a: String = "100"
val b: String = "9"
val result = operation match
case '+' => (a.toInt + b.toInt).toString
case '-' => (a.toInt - b.toInt).toString
case '*' => (a.toInt * b.toInt).toString
case '/' if b.toInt != 0 => (a.toInt / b.toInt).toString
case '/' => "Деление на ноль!"
case '%' if b.toInt != 0 => (a.toInt % b.toInt).toString
case '%' => "Остаток от деления на ноль!"
case '^' => math.pow(a.toInt, b.toInt).toString
case op => s"Неизвестная операция: '$op'"

result // = "109"
